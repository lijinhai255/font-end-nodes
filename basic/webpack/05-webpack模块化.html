# жЁЎеқ—еҢ–и§„иҢғ

дёүеӨ§ JavaScript дё»жөҒжЁЎеқ—и§„иҢғпјҡCommonJSгҖҒAMD е’Ң ES6 ModuleгҖӮ

# CommonJS

CommonJS (opens new window) и§„иҢғжҳҜ 2009 е№ҙдёҖеҗҚжқҘиҮӘ Mozilla еӣўйҳҹзҡ„е·ҘзЁӢеёҲ Kevin Dangoor ејҖе§Ӣи®ҫи®ЎдёҖдёӘеҸ« ServerJS зҡ„йЎ№зӣ®жҸҗеҮәжқҘзҡ„пјҢйҡҸзқҖ Node.js зҡ„е№ҝжіӣеә”з”ЁпјҢиў«е№ҝжіӣжҺҘеҸ—гҖӮйҖҡиҝҮ ServerJS иҝҷдёӘеҗҚеӯ—е°ұеҸҜд»ҘзҹҘйҒ“пјҢCommonJS дё»иҰҒжҳҜжңҚеҠЎз«Ҝз”Ёзҡ„жЁЎеқ—и§„иҢғгҖӮ

// sayhi.js
var hi = 'hello world';

function sayHi() {
    return hi;
}

module.exports = sayHi;
// index.js

var sayHi = require('./sayhi.js');

console.log(sayHi());

дёҠйқўзҡ„д»Јз Ғе°ұжҳҜ CommonJS иҜӯжі•пјҢдҪҝз”ЁдәҶrequireжқҘеҜје…ҘдёҖдёӘжЁЎеқ—пјҢmodule.exportsеҜјеҮәжЁЎеқ—гҖӮеңЁ Node.js дёӯе®һйҷ…д»Јз Ғ (opens new window)дјҡиў«еӨ„зҗҶжҲҗдёӢйқўд»Јз ҒиҖҢиў«еә”з”Ёпјҡ

(function(exports, require, module, __filename, __dirname) {
    // ...
    // жЁЎеқ—зҡ„д»Јз ҒеңЁиҝҷйҮҢ
    // ...
});

CommonJS зҡ„й—®йўҳпјҡCommonjsи§„иҢғжҳҜдёҚйҖӮеҗҲжөҸи§ҲеҷЁзҡ„пјҢдҪҶз”ұдәҺжңүдәҶжү“еҢ…е·Ҙе…·зҡ„ж”Ҝж’‘пјҢйҖҡиҝҮеӨ„зҗҶзҡ„Commonjsд№ҹеҸҜд»ҘеңЁжөҸи§ҲеҷЁдёӯдҪҝз”ЁгҖӮ

CommonJS и§„иҢғжҳҜ JavaScript дёӯжңҖеёёи§Ғзҡ„жЁЎеқ—ж јејҸи§„иҢғпјҢиҝҷдёҖж ҮеҮҶзҡ„и®ҫи®ЎеҲқиЎ·жҳҜдёәдәҶи®© JavaScript еңЁеӨҡдёӘзҺҜеўғдёӢйғҪе®һзҺ°жЁЎеқ—еҢ–гҖӮ

иө·е…Ҳдё»иҰҒеә”з”ЁеңЁ Node.js жңҚеҠЎз«ҜдёӯпјҢдҪҶжҳҜ Node.js дёӯзҡ„е®һзҺ°дҫқиө–дәҶ Node.js жң¬иә«еҠҹиғҪзҡ„е®һзҺ°пјҢеҢ…жӢ¬дәҶ Node.js зҡ„ж–Үд»¶зі»з»ҹзӯүпјҢиҝҷдёӘи§„иҢғеңЁжөҸи§ҲеҷЁзҺҜеўғжҳҜжІЎжі•дҪҝз”Ёзҡ„гҖӮ

еҗҺжқҘйҡҸзқҖ Browserify (opens new window) е’Ң Webpack зӯүжү“еҢ…е·Ҙе…·зҡ„еҙӣиө·пјҢйҖҡиҝҮеӨ„зҗҶзҡ„ CommonJS еүҚз«Ҝд»Јз Ғд№ҹеҸҜд»ҘеңЁжөҸи§ҲеҷЁдёӯдҪҝз”ЁгҖӮ

# AMD

AMD и§„иҢғ (opens new window)жҳҜеңЁ CommonJS и§„иҢғд№ӢеҗҺжҺЁеҮәзҡ„дёҖдёӘи§ЈеҶі web йЎөйқўеҠЁжҖҒејӮжӯҘеҠ иҪҪ JavaScript зҡ„и§„иҢғпјҢзӣёеҜ№дәҺ CommonJS и§„иҢғпјҢе®ғжңҖеӨ§зү№зӮ№жҳҜжөҸи§ҲеҷЁеҶ…ж”ҜжҢҒгҖҒе®һзҺ°з®ҖеҚ•гҖҒе№¶дё”ж”ҜжҢҒејӮжӯҘеҠ иҪҪпјҢAMD и§„иҢғжңҖж—©жҳҜйҡҸзқҖRequireJS (opens new window)еҸ‘еұ•иҖҢжҸҗеҮәжқҘзҡ„пјҢе®ғжңҖж ёеҝғзҡ„жҳҜдёӨдёӘж–№жі•пјҡ

  • require()пјҡеј•е…Ҙе…¶д»–жЁЎеқ—пјӣ
  • define()пјҡе®ҡд№үж–°зҡ„жЁЎеқ—гҖӮ

еҹәжң¬иҜӯжі•еҰӮдёӢпјҡ

// sayhi.js
define(function() {
    var hi = 'hello world';
    return function sayHi() {
        return hi;
    };
});

// index.js
require(['./sayhi.js'], function(sayHi) {
    console.log(sayHi());
});

AMD жҸҗеҮәжқҘд№ӢеҗҺпјҢд№ҹжңүеҫҲеӨҡеҸҳз§Қзҡ„и§„иҢғжҸҗеҮәжқҘпјҢжҜ”еҰӮеӣҪеҶ… Sea.js (opens new window) зҡ„ CMD (opens new window)пјҢиҝҳжңүе…је®№ CommonJS е’Ң AMD зҡ„ UMD и§„иҢғ (opens new window)пјҲUniversal Module DefinitionпјүгҖӮиҷҪз„¶ AMD зҡ„жЁЎејҸеҫҲйҖӮеҗҲжөҸи§ҲеҷЁз«Ҝзҡ„ејҖеҸ‘пјҢдҪҶжҳҜйҡҸзқҖ npm еҢ…з®ЎзҗҶзҡ„жңәеҲ¶и¶ҠжқҘи¶ҠжөҒиЎҢпјҢиҝҷз§Қж–№ејҸеҸҜиғҪдјҡйҖҗжӯҘзҡ„иў«ж·ҳжұ°жҺүгҖӮ

AMD и§„иҢғзҡ„й—®йўҳпјҡдҫқиө–еүҚзҪ®еҠ еӨ§дәҶејҖеҸ‘зҡ„йҡҫеәҰпјҢж— и®әеңЁйҳ…иҜ»д»Јз ҒиҝҳжҳҜзј–еҶҷд»Јз Ғзҡ„ж—¶пјҢйғҪдјҡеҜјиҮҙеј•е…Ҙзҡ„жЁЎеқ—еҶ…е®№жҳҜжқЎд»¶жҖ§жү§иЎҢзҡ„гҖӮ

еңЁ AMD и§„иҢғдёӯпјҢиҰҒеЈ°жҳҺдёҖдёӘжЁЎеқ—пјҢйӮЈд№ҲйңҖиҰҒжҢҮе®ҡиҜҘжЁЎеқ—з”ЁеҲ°зҡ„жүҖжңүдҫқиө–йЎ№пјҢиҝҷдәӣдҫқиө–йЎ№дјҡиў«еҪ“еҒҡеҪўеҸӮдј еҲ° factoryпјҲdefineж–№жі•дј е…Ҙзҡ„еҮҪж•°еҸ«еҒҡfactoryпјүдёӯпјҢеҜ№дәҺдҫқиө–зҡ„жЁЎеқ—дјҡжҸҗеүҚжү§иЎҢпјҢиҝҷз§ҚеҒҡжі•еҸ«еҒҡ дҫқиө–еүҚзҪ®гҖӮдҫқиө–еүҚзҪ®еҠ еӨ§дәҶејҖеҸ‘зҡ„йҡҫеәҰпјҢж— и®әеңЁйҳ…иҜ»д»Јз ҒиҝҳжҳҜзј–еҶҷд»Јз Ғзҡ„ж—¶пјҢйғҪдјҡеҜјиҮҙеј•е…Ҙзҡ„жЁЎеқ—еҶ…е®№жҳҜжқЎд»¶жҖ§жү§иЎҢзҡ„гҖӮ

иҖҢдё”дёҚз®Ў AMD иҝҳжҳҜ CommonJS йғҪжІЎжңүз»ҹдёҖжөҸи§ҲеҷЁе’Ңе®ўжҲ·з«Ҝзҡ„жЁЎеқ—еҢ–и§„иҢғгҖӮ

# ES6 Module

ES6 ModuleпјҢеҸҲз§°дёә ES2015 modulesпјҢжҳҜ ES2015 ж ҮеҮҶжҸҗеҮәжқҘзҡ„дёҖз§ҚжЁЎеқ—еҠ иҪҪж–№ејҸпјҢд№ҹжҳҜ ECMAScript е®ҳж–№жҸҗеҮәзҡ„ж–№жЎҲпјҢдҪңдёә ES ж ҮеҮҶпјҢдёҚд»…д»…еңЁ Web зҺ°д»ЈжөҸи§ҲеҷЁпјҲдҫӢеҰӮ ChromeпјүдёҠйқўеҫ—еҲ°е®һзҺ°пјҢиҖҢдё”еңЁ Node.js 9+ зүҲжң¬д№ҹеҫ—еҲ°еҺҹз”ҹж”ҜжҢҒпјҲйңҖиҰҒеҠ дёҠ--experimental-modulesдҪҝз”ЁпјүгҖӮ

// sayhi.js
const hi = 'hello world';
export default function sayHi() {
    return hi;
}

// index.js
import sayHi from './sayhi';
console.log(sayHi());

еҜ№дәҺеүҚз«ҜйЎ№зӣ®пјҢеҸҜд»ҘйҖҡиҝҮ Babel жҲ–иҖ… Typescript иҝӣиЎҢжҸҗеүҚдҪ“йӘҢгҖӮ

йҡҸзқҖдё»жөҒжөҸи§ҲеҷЁйҖҗжӯҘејҖе§Ӣж”ҜжҢҒ ES ModulesпјҲESMпјү ж ҮеҮҶпјҢи¶ҠжқҘи¶ҠеӨҡзҡ„зӣ®е…үжҠ•жіЁдәҺ Node.js еҜ№дәҺ ESM зҡ„ж”ҜжҢҒе®һзҺ°дёҠгҖӮ

зӣ®еүҚ Node.js дҪҝз”Ё CommonJS дҪңдёәе®ҳж–№зҡ„жЁЎеқ—и§ЈеҶіж–№жЎҲпјҢиҷҪз„¶еҶ…зҪ®зҡ„жЁЎеқ—ж–№жЎҲдҝғиҝӣдәҶ Node.js зҡ„жөҒиЎҢпјҢдҪҶжҳҜд№ҹдёәеј•е…Ҙж–°зҡ„ ES Modules йҖ жҲҗдәҶдёҖе®ҡзҡ„йҳ»зўҚгҖӮдёҚиҝҮ Node.js 9.0+ е·Із»Ҹж”ҜжҢҒ ESM иҜӯжі•пјҢзңӢзӨәдҫӢпјҡ

// sayhi.mjs
const hi = 'hello world';
export default function sayHi() {
    return hi;
}

// index.mjs
import sayHi from './sayhi.mjs';
console.log(sayHi());

// дҪҝз”Ёе‘Ҫд»Ө
вһң node --experimental-modules index.mjs
hello world

йңҖиҰҒж·»еҠ  flag --experimental-modules жқҘеҗҜеҠЁ ESM иҜӯжі•ж”ҜжҢҒпјҢж–Үд»¶еҲҷеҝ…йЎ»дҪҝз”Ё .mjs еҗҺзјҖпјҡ node --experimental-modules some-esm-file.mjs гҖӮ

# Webpackе·ҘдҪңеҺҹзҗҶз®Җжһҗ

еңЁ Web еүҚз«ҜпјҢдёҚд»…д»…еҸӘжңү JavaScriptпјҢиҝҳжңү CSSгҖҒHTMLгҖҒеӣҫзүҮгҖҒеӯ—дҪ“гҖҒеҜҢеӘ’дҪ“зӯүдј—еӨҡиө„жәҗпјҢиҝҳжңүдёҖдәӣиө„жәҗжҳҜд»Ҙзұ»дјјгҖҢж–№иЁҖгҖҚзҡ„ж–№ејҸеӯҳеңЁзқҖпјҢжҜ”еҰӮ lessгҖҒsassгҖҒеҗ„з§Қ js жЁЎжқҝеә“зӯүпјҢиҝҷдәӣиө„жәҗе№¶дёҚиғҪиў«зӣҙжҺҘз”ЁеңЁ JavaScript дёӯпјҢеҰӮжһңеңЁ JavaScript дёӯеғҸдҪҝз”ЁжЁЎеқ—дёҖж ·дҪҝз”ЁпјҢйӮЈд№ҲеҸҜд»ҘжһҒеӨ§зҡ„жҸҗй«ҳзҡ„ејҖеҸ‘дҪ“йӘҢпјҡ

var img = require('./img/webpack.png');
var style = require('./css/style.css');
var template = require('./template.ejs');

иҝҷж—¶еҖҷпјҢе°ұйңҖиҰҒ Webpack дәҶпјҒеңЁ Webpack дёӯпјҢдёҖеҲҮзҡҶжЁЎеқ—пјҒ

webpackзҡ„зј–иҜ‘йҖ»иҫ‘пјҡ

  • Webpack дјҡиҰҒеҜ№ж•ҙдёӘд»Јз ҒиҝӣиЎҢйқҷжҖҒеҲҶжһҗпјҢеҲҶжһҗеҮәеҗ„дёӘжЁЎеқ—зҡ„зұ»еһӢе’Ңе®ғд»¬дҫқиө–е…ізі»
  • е°ҶдёҚеҗҢзұ»еһӢзҡ„жЁЎеқ—жҸҗдәӨз»ҷеҜ№еә”зҡ„еҠ иҪҪеҷЁпјҲloaderпјүжқҘеӨ„зҗҶгҖӮ

жҜ”еҰӮпјҡдёҖдёӘз”Ё Less еҶҷзҡ„ж ·ејҸпјҢеҸҜд»Ҙе…Ҳз”Ё less-loader е°Ҷе®ғиҪ¬жҲҗдёҖдёӘ CSS жЁЎеқ—пјҢз„¶еҗҺеҶҚйҖҡиҝҮ css-loader жҠҠд»–жҸ’е…ҘеҲ°йЎөйқўзҡ„ <style> ж Үзӯҫдёӯжү§иЎҢпјҢз”ҡиҮіиҝҳеҸҜд»ҘйҖҡиҝҮжҸ’д»¶е°ҶиҝҷйғЁеҲҶ CSS еҜјеҮәдёә CSS ж–Үд»¶пјҢдҪҝз”Ёlinkж Үзӯҫеј•е…ҘеҲ°йЎөйқўдёӯгҖӮ

# webpackеҜ№Module зҡ„еўһејә

еңЁ Webpack дёӯпјҢдёҚд»…еҸҜд»ҘдёәжүҖж¬Ідёәзҡ„дҪҝз”Ё CommonJS гҖҒAMD е’Ң ES6 Module дёүеӨ§и§„иҢғпјҲжҜ”еҰӮдёҖдёӘж–Үд»¶дёӯж··еҗҲдҪҝз”Ёдёүз§Қи§„иҢғпјүпјҢиҝҳеҸҜд»ҘдҪҝз”Ё Webpack еҜ№ Module зҡ„еўһејәж–№жі•е’ҢеұһжҖ§гҖӮ

# import()е’ҢзҘһеҘҮжіЁйҮҠ

еңЁ Webpack дёӯпјҢimportдёҚд»…д»…жҳҜ ES6 Module зҡ„жЁЎеқ—еҜје…Ҙж–№ејҸпјҢиҝҳжҳҜдёҖдёӘзұ»дјјrequireзҡ„еҮҪж•°пјҲе…¶е®һиҝҷжҳҜ ES2015 loader и§„иҢғ (opens new window)зҡ„е®һзҺ°пјүпјҢеҸҜд»ҘйҖҡиҝҮimport('path/to/module')зҡ„ж–№ејҸеј•е…ҘдёҖдёӘжЁЎеқ—пјҢimport()иҝ”еӣһзҡ„жҳҜдёҖдёӘPromiseеҜ№иұЎпјҡ

import('path/to/module').then(mod => {
    console.log(mod);
});

еҸӮиҖғе…ҲеүҚзҡ„дёҖз« пјҢжқҘеҲӣе»әеҹәзЎҖзҡ„ES6зҡ„webpackйЎ№зӣ®гҖӮ

дёӢйқўзңӢзңӢдҪҝз”Ёimport()е’Ң import зҡ„жү“еҢ…жңүд»Җд№ҲеҢәеҲ«пјҡ

// hello.js
export default 'hello';

// lazy.js
export default 'lazy module';

// index.js
import hello from './hello';
import('./lazy').then(lazy => {
    console.log(lazy);
});

npx webpack --mode developmentпјҡ

йҖҡиҝҮжү“еҢ…еҗҺзҡ„ log е’Ңdistж–Үд»¶еӨ№еҶ…е®№еҸ‘зҺ°пјҢзҡ„д»Јз Ғиў«еҲҶеүІжҲҗдәҶдёӨдёӘж–Үд»¶пјҢдёҖдёӘжҳҜmain.jsдёҖдёӘжҳҜlazy_js.jsпјҢиҝҷжҳҜеӣ дёәзӣёеҜ№дәҺimport fromзҡ„йқҷжҖҒеҲҶжһҗжү“еҢ…иҜӯжі•пјҢimport()жҳҜеҠЁжҖҒжү“еҢ…иҜӯжі•пјҢеҚізҡ„еҶ…е®№дёҚжҳҜз¬¬дёҖж—¶й—ҙжү“иҝӣmain.jsзҡ„пјҢиҖҢжҳҜйҖҡиҝҮејӮжӯҘзҡ„ж–№ејҸеҠ иҪҪиҝӣжқҘзҡ„гҖӮд»Јз ҒеҲҶеүІжҳҜ webpack иҝӣиЎҢд»Јз Ғз»“жһ„з»„з»ҮпјҢе®һзҺ°еҠЁжҖҒдјҳеҢ–зҡ„дёҖдёӘйҮҚиҰҒеҠҹиғҪ

дёҺimport()з”Ёжі•дёҖж ·зҡ„жҳҜrequire.ensureзҡ„ж–№жі•пјҢиҝҷдёӘж–№жі•е·Із»Ҹиў«import()ж–№ејҸжӣҝжҚўжҺүпјӣ

й’ҲеҜ№import()жү“еҢ…дә§зү©и·ҹжҷ®йҖҡзҡ„йқҷжҖҒеҲҶжһҗжү“еҢ…зҡ„е®һзҺ°дёҚеҗҢд№ӢеӨ„пјҢеҗҺйқўеҺҹзҗҶзҜҮи®Іи§Јжү“еҢ…дә§еҮәзү©зҡ„ж—¶еҖҷдјҡиҜҰз»Ҷд»Ӣз»ҚгҖӮ

дёӢйқўеҶҚжқҘзңӢдёӢimport()зҡ„зҘһеҘҮжіЁйҮҠзү№жҖ§пјҢдёҠйқўindex.jsзҡ„д»Јз Ғдҝ®ж”№жҲҗдёӢйқўиҝҷж ·пјҢеўһеҠ жіЁйҮҠwebpackChunkName: 'lazy-name'

import hello from './hello';
import(
    /*
     webpackChunkName: 'toimc-lazy'
     */
    './lazy'
).then(lazy => {
    console.log(lazy);
});

еҲҷжү“еҢ…еҗҺзҡ„з»“жһңпјҢlasy_js.jsеҸҳжҲҗдәҶtoimc-lazy.jsдәҶпјҢиҝҷдёӘж–Үд»¶зҡ„еҗҚеӯ—е°ұжҳҜеңЁimport()жіЁйҮҠйҮҢйқўжҢҮе®ҡзҡ„webpackChunkNameпјҢиҝҷе°ұжҳҜзҘһеҘҮжіЁйҮҠпјҲMagic CommentsпјүгҖӮ

зӣ®еүҚж”ҜжҢҒзҡ„жіЁйҮҠжңүпјҡ

  • webpackIncludeпјҡеҰӮжһңжҳҜ import зҡ„дёҖдёӘзӣ®еҪ•пјҢеҲҷеҸҜд»ҘжҢҮе®ҡйңҖиҰҒеј•е…Ҙзҡ„ж–Үд»¶зү№жҖ§пјҢдҫӢеҰӮеҸӘеҠ иҪҪ json ж–Үд»¶пјҡ/\.json$/пјӣ
  • webpackExcludeпјҡеҰӮжһңжҳҜ import зҡ„дёҖдёӘзӣ®еҪ•пјҢеҲҷеҸҜд»ҘжҢҮе®ҡйңҖиҰҒиҝҮж»Өзҡ„ж–Үд»¶пјҢдҫӢеҰӮ /\.noimport\.json$/пјӣ
  • webpackChunkNameпјҡиҝҷжҳҜ chunk ж–Үд»¶зҡ„еҗҚз§°пјҢдҫӢеҰӮ toimc-lazyпјӣ
  • webpackPrefetch: жҳҜеҗҰйў„еҸ–жЁЎеқ—пјҢеҸҠе…¶дјҳе…Ҳзә§пјҢеҸҜйҖүеҖјtrueгҖҒжҲ–иҖ…ж•ҙж•°дјҳе…Ҳзә§еҲ«пјҢ0 зӣёеҪ“дәҺ trueпјҢwebpack 4.6+ж”ҜжҢҒпјӣ
  • webpackPreload: жҳҜеҗҰйў„еҠ иҪҪжЁЎеқ—пјҢеҸҠе…¶дјҳе…Ҳзә§пјҢеҸҜйҖүеҖјtrueгҖҒжҲ–иҖ…ж•ҙж•°дјҳе…Ҳзә§еҲ«пјҢ0 зӣёеҪ“дәҺ trueпјҢwebpack 4.6+ж”ҜжҢҒпјӣ
  • webpackMode: еҸҜйҖүеҖјlazy/lazy-once/eager/weakгҖӮ
import hello from './hello';

console.log(hello)

import(
  /* webpackChunkName: 'toimc-lazy' */
  /* webpackInclude: /\.json$/ */
  /* webpackExclude: /\.noimport\.json$/ */
  /* webpackMode: "lazy" */
  /* webpackPrefetch: true */
  /* webpackPreload: true */
  './lazy').then(lazy => {
    console.log(lazy);
});

иҝҷйҮҢжңҖеӨҚжқӮзҡ„жҳҜwebpackModeпјҡ

  • lazyпјҡжҳҜй»ҳи®Өзҡ„жЁЎејҸпјҢдёәжҜҸдёӘ import() еҜје…Ҙзҡ„жЁЎеқ—пјҢз”ҹжҲҗдёҖдёӘеҸҜе»¶иҝҹеҠ иҪҪ chunkпјӣ
  • lazy-onceпјҡз”ҹжҲҗдёҖдёӘеҸҜд»Ҙж»Ўи¶іжүҖжңү import() и°ғз”Ёзҡ„еҚ•дёӘеҸҜе»¶иҝҹеҠ иҪҪ chunkпјҢжӯӨ chunk е°ҶеңЁз¬¬дёҖж¬Ў import() и°ғз”Ёж—¶иҺ·еҸ–пјҢйҡҸеҗҺзҡ„ import() и°ғз”Ёе°ҶдҪҝз”ЁзӣёеҗҢзҡ„зҪ‘з»ңе“Қеә”пјӣжіЁж„ҸпјҢиҝҷз§ҚжЁЎејҸд»…еңЁйғЁеҲҶеҠЁжҖҒиҜӯеҸҘдёӯжңүж„Ҹд№үпјҢдҫӢеҰӮ import(./locales/${language}.json)пјҢе…¶дёӯеҸҜиғҪеҗ«жңүеӨҡдёӘиў«иҜ·жұӮзҡ„жЁЎеқ—и·Ҝеҫ„пјӣ
  • eagerпјҡдёҚдјҡз”ҹжҲҗйўқеӨ–зҡ„ chunkпјҢжүҖжңүжЁЎеқ—йғҪиў«еҪ“еүҚ chunk еј•е…ҘпјҢе№¶дё”жІЎжңүйўқеӨ–зҡ„зҪ‘з»ңиҜ·жұӮгҖӮд»Қз„¶дјҡиҝ”еӣһ PromiseпјҢдҪҶжҳҜжҳҜ resolved зҠ¶жҖҒгҖӮе’ҢйқҷжҖҒеҜје…ҘзӣёеҜ№жҜ”пјҢеңЁи°ғз”Ё import() е®ҢжҲҗд№ӢеүҚпјҢиҜҘжЁЎеқ—дёҚдјҡиў«жү§иЎҢгҖӮ
  • weakпјҡе°қиҜ•еҠ иҪҪжЁЎеқ—пјҢеҰӮжһңиҜҘжЁЎеқ—еҮҪж•°е·Із»Ҹд»Ҙе…¶д»–ж–№ејҸеҠ иҪҪпјҲеҚіпјҢеҸҰдёҖдёӘ chunk еҜје…ҘиҝҮжӯӨжЁЎеқ—пјҢжҲ–еҢ…еҗ«жЁЎеқ—зҡ„и„ҡжң¬иў«еҠ иҪҪпјүгҖӮд»Қз„¶дјҡиҝ”еӣһ PromiseпјҢдҪҶжҳҜеҸӘжңүеңЁе®ўжҲ·з«ҜдёҠе·Із»ҸжңүиҜҘ chunk ж—¶жүҚжҲҗеҠҹи§ЈжһҗгҖӮеҰӮжһңиҜҘжЁЎеқ—дёҚеҸҜз”ЁпјҢPromise е°ҶдјҡжҳҜ rejected зҠ¶жҖҒпјҢе№¶дё”зҪ‘з»ңиҜ·жұӮж°ёиҝңдёҚдјҡжү§иЎҢгҖӮеҪ“йңҖиҰҒзҡ„ chunks е§Ӣз»ҲеңЁпјҲеөҢе…ҘеңЁйЎөйқўдёӯзҡ„пјүеҲқе§ӢиҜ·жұӮдёӯжүӢеҠЁжҸҗдҫӣпјҢиҖҢдёҚжҳҜеңЁеә”з”ЁзЁӢеәҸеҜјиҲӘеңЁжңҖеҲқжІЎжңүжҸҗдҫӣзҡ„жЁЎеқ—еҜје…Ҙзҡ„жғ…еҶөи§ҰеҸ‘пјҢиҝҷеҜ№дәҺ Server з«ҜжёІжҹ“пјҲSSRпјҢServer-Side RenderпјүжҳҜйқһеёёжңүз”Ёзҡ„гҖӮ

йҖҡиҝҮдёҠйқўзҡ„зҘһеҘҮжіЁйҮҠпјҢimport()дёҚеҶҚжҳҜз®ҖеҚ•зҡ„ JavaScript ејӮжӯҘеҠ иҪҪеҷЁпјҢиҝҳжҳҜд»»ж„ҸжЁЎеқ—иө„жәҗзҡ„еҠ иҪҪеҷЁпјҢдёҫдҫӢиҜҙжҳҺпјҡеҰӮжһңйЎөйқўз”ЁеҲ°зҡ„еӣҫзүҮйғҪж”ҫеңЁsrc/assets/imgж–Үд»¶еӨ№дёӢпјҢдҪ д»¬еҸҜд»ҘйҖҡиҝҮдёӢйқўж–№ејҸе°Ҷз”ЁеҲ°зҡ„еӣҫзүҮжү“еҢ…еҲ°дёҖиө·пјҡ

import(/* webpackChunkName: "image", webpackInclude: /\.(png|jpg|gif)/ */ './assets/img');

prefetch дјҳе…Ҳзә§дҪҺдәҺ preloadпјҢpreload дјҡе№¶иЎҢжҲ–иҖ…еҠ иҪҪе®Ңдё»ж–Үд»¶д№ӢеҗҺз«ӢеҚіеҠ иҪҪпјӣprefetch еҲҷдјҡеңЁдё»ж–Үд»¶д№ӢеҗҺгҖҒз©әй—Іж—¶еңЁеҠ иҪҪгҖӮprefetch е’Ң preload еҸҜд»Ҙз”ЁдәҺжҸҗеүҚеҠ иҪҪеӣҫзүҮгҖҒж ·ејҸзӯүиө„жәҗзҡ„еҠҹиғҪгҖӮ

# require.resolve() е’Ңrequire.resolveWeak()

require.resolve() е’Ң require.resolveWeak()йғҪеҸҜд»ҘиҺ·еҸ–жЁЎеқ—зҡ„е”ҜдёҖ IDпјҲmoduleIdпјүпјҢеҢәеҲ«еңЁдәҺrequire.resolve()дјҡжҠҠжЁЎеқ—зңҹе®һеј•е…Ҙиҝӣ bundleпјҢиҖҢrequire.resolveWeak()еҲҷдёҚдјҡпјҢй…ҚеҗҲrequire.cacheе’Ң __webpack_modules__еҸҜд»Ҙз”ЁдәҺеҲӨж–ӯжЁЎеқ—жҳҜеҗҰеҠ иҪҪжҲҗеҠҹжҲ–иҖ…жҳҜеҗҰеҸҜз”ЁгҖӮ

if (__webpack_modules__[require.resolveWeak('module')]) {
  // еҪ“жЁЎеқ—еҸҜз”Ёж—¶пјҢеҒҡдёҖдәӣдәӢжғ…...
}
if (require.cache[require.resolveWeak('module')]) {
  // еҪ“жЁЎеқ—иў«еҠ иҪҪд№ӢеүҚпјҢеҒҡдёҖдәӣдәӢжғ…...
}

// еҸҜд»Ҙжү§иЎҢеҠЁжҖҒи§ЈжһҗпјҲ"context"пјү
// дёҺе…¶д»– require/import ж–№жі•зұ»дјјгҖӮ
const page = 'Foo';
__webpack_modules__[require.resolveWeak(`./page/${page}`)];

# require.context()

require.context(directory, includeSubdirs, filter)еҸҜд»Ҙжү№йҮҸе°ҶdirectoryеҶ…зҡ„ж–Үд»¶е…ЁйғЁеј•е…Ҙиҝӣж–Үд»¶(иҝҷдёӘеңЁи§Ҷйў‘дёӯжңүд»Ӣз»ҚеҲ°)пјҢе№¶дё”иҝ”еӣһдёҖдёӘе…·жңүresolveзҡ„ context еҜ№иұЎпјҢдҪҝз”Ёcontext.resolve(moduleId)еҲҷиҝ”еӣһеҜ№еә”зҡ„жЁЎеқ—гҖӮ

  • directoryпјҡзӣ®еҪ• stringпјӣ
  • includeSubdirsпјҡжҳҜеҗҰеҢ…еҗ«еӯҗзӣ®еҪ•пјҢеҸҜйҖүпјҢй»ҳи®ӨеҖјжҳҜ trueпјӣ
  • filterпјҡиҝҮж»ӨжӯЈеҲҷи§„еҲҷпјҢеҸҜйҖүйЎ№гҖӮ

TipsпјҡжіЁж„Ҹ require.context() дјҡе°ҶжүҖжңүзҡ„ж–Үд»¶йғҪеј•е…Ҙиҝӣ bundleпјҒ

# require.include()

require.include()е·Іиў«еәҹејғпјҢдёҚд№…е°Ҷиў«еҲ йҷӨгҖӮ

require.include(dependency)йЎҫеҗҚжҖқд№үдёәеј•е…ҘжҹҗдёӘдҫқиө–пјҢдҪҶжҳҜе№¶дёҚжү§иЎҢе®ғпјҢеҸҜд»Ҙз”ЁдәҺдјҳеҢ– chunkпјҢдҫӢеҰӮдёӢйқўзӨәдҫӢд»Јз Ғпјҡ

require.include('./hello.js');
require.ensure(['./hello.js', './weak.js'], function(require) {
    /* ... */
});
require.ensure(['./hello.js', './lazy.js'], function(require) {
    /* ... */
});

# иө„жәҗзҡ„жЁЎеқ—еҢ–еӨ„зҗҶ

Webpack иҝҳеҜ№дёҖдәӣеёёз”Ёзҡ„ Node.js жЁЎеқ—е’ҢеұһжҖ§иҝӣиЎҢдәҶ mockпјҢдҫӢеҰӮпјҡеңЁ web зҡ„ js ж–Үд»¶дёӯеҸҜд»ҘзӣҙжҺҘеј•е…Ҙ Node.js зҡ„querystringжЁЎеқ—пјҢиҝҷдёӘжЁЎеқ—е®һйҷ…еј•е…Ҙзҡ„жҳҜnode-libs-browser (opens new window)жқҘеҜ№ Node.js ж ёеҝғеә“ polyfillпјҢиҜҰз»ҶеңЁ web йЎөйқўдёӯеҸҜд»Ҙз”Ёзҡ„ Node.js жЁЎеқ—пјҢеҸҜд»ҘеҸӮиҖғnode-libs-browser (opens new window) Readme ж–Үд»¶зҡ„иЎЁж јгҖӮ

# @importе’Ңimport

еңЁ Webpack дёӯзҡ„ css пјҲеҢ…жӢ¬е…¶йў„еӨ„зҗҶиҜӯиЁҖпјҢдҫӢеҰӮ LessгҖҒSassпјүзӯүпјҢйғҪеҸҜд»ҘеңЁеҶ…йғЁйҖҡиҝҮ@importиҜӯжі•зӣҙжҺҘеј•е…ҘдҪҝз”Ёпјҡ

@import 'vars.less';
body {
    background: @bg-color;
}

йҷӨдәҶж ·ејҸж–Үд»¶дёӯзҡ„@importпјҢеңЁ JavaScript ж–Үд»¶дёӯпјҢиҝҳж”ҜжҢҒзӣҙжҺҘдҪҝз”Ё ES6 Module зҡ„importпјҲеҢ…жӢ¬ requireпјүзӣҙжҺҘеј•е…Ҙж ·ејҸж–Үд»¶пјҢдҫӢеҰӮпјҡ

import styles from './style.css';

JavaScript зҡ„иҝҷз§ҚиҜӯжі•е…¶е®һжҳҜ CSS Modules иҜӯжі• (opens new window)пјҢзӣ®еүҚжөҸи§ҲеҷЁж”ҜжҢҒзЁӢеәҰжңүйҷҗпјҢдҪҶжҳҜеңЁ Webpack дёӯпјҢеҸҜд»ҘйҖҡиҝҮй…ҚзҪ® loader дјҳе…ҲдҪҝз”Ёиҝҷз§Қж–№ејҸпјҒ

еҗҺйқўи®Іи§Ј CSS ж ·ејҸй…ҚзҪ®зҡ„ж—¶еҖҷдјҡжӣҙеҠ иҜҰз»Ҷзҡ„и®Іи§Ј CSS Modules

# жҠҠиө„жәҗдҪңдёәжЁЎеқ—еј•е…Ҙ

еңЁ Webpack дёӯпјҢйҷӨдәҶ CSS еҸҜд»ҘзӣҙжҺҘдҪҝз”ЁimportиҜӯжі•еј•е…ҘпјҢзұ»дјј HTML е’ҢйЎөйқўжЁЎжқҝзӯүпјҢеҸҜд»ҘзӣҙжҺҘдҪҝз”ЁеҜ№еә”зҡ„ loader пјҢйҖҡиҝҮдёӢйқўзҡ„иҜӯжі•жқҘеј•е…Ҙпјҡ

const html = require('html-loader!./loader.html');
console.log(html);

дёҠйқўзҡ„д»Јз Ғеҫ—еҲ°htmlеҸҳйҮҸе®һйҷ…дёәеј•е…Ҙзҡ„loader.htmlзҡ„ string зүҮж®өгҖӮйҷӨдәҶ htmlпјҢзұ»дјјжЁЎжқҝж–Үд»¶пјҢиҝҳеҸҜд»ҘзӣҙжҺҘиҪ¬жҚўдёәеҜ№еә”зҡ„ render еҮҪж•°пјҢдҫӢеҰӮдёӢйқўд»Јз ҒдҪҝз”ЁдәҶejs-loader (opens new window)пјҡ

const render = require('ejs!./file.ejs');
// => еҫ—еҲ° ejs зј–иҜ‘еҗҺзҡ„ render еҮҪж•°
render(data); // дј е…Ҙ dataпјҢзӣҙжҺҘиҝ”еӣһзҡ„жҳҜ html зүҮж®ө

еҺҹзҗҶзҜҮе°ҶеҠЁжүӢжүӢеҶҷдёҖдёӘ markdown-loader пјҢжӣҙж·ұе…Ҙзҡ„дәҶи§Је…¶еҠҹиғҪе’ҢеҺҹзҗҶе®һзҺ°гҖӮ

еҗҺз»ӯзҡ„дёҖз« пјҢе°Ҷдјҡд»Ӣз»ҚwebpackдёӯйҮҚиҰҒзҡ„loadersеӨ„зҗҶпјҢwebpackеҖҹеҠ©loadersжқҘеӨ„зҗҶеҗ„ејҸеҗ„ж ·зҡ„иө„жәҗгҖӮ
