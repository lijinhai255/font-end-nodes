# 有效括号

栈

/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    const stack = [];
    const pairs = {
        '(':')',
        "[":']',
        '{':'}'
    };

    for(let i = 0;i<s.length;i++){
        const char = s[i];
        if(char in pairs){
            // 如果存在
            stack.push(char)
        }else{
            // key 根本不在pairs中  ) ] } false
            // pop 不能合当前的char 匹配成一对
            if(stack.length===0||pairs[stack.pop()]!==char){
                return false 
            }
        }
    }
    return stack.length ===0

};
